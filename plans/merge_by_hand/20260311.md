```python
class HorizontalStitch:
    """
    横图拼接算法类

    使用方式:
        stitcher = HorizontalStitch(task_id, conf)
        flag, details = stitcher.process()
    """

    def __init__(self, task_id: str, conf: dict):
        self.task_id = task_id
        self.conf = conf
        self.rib_count = conf.get('rib_count', 5)
        self.symmetry_type = conf.get('symmetry_type', 'asymmetric')
        self.center_images = []
        self.side_images = []
        self.image_names = {}
        self.results = []

    def process(self) -> Tuple[bool, dict]:
        try:
            # 1. 加载图片
            self.center_images, self.side_images, self.image_names = load_images_from_directories(self.conf)

            if not self.center_images or not self.side_images:
                return False, {
                    "error": "图片加载失败",
                    "center_count": len(self.center_images),
                    "side_count": len(self.side_images)
                }

            # 2. 生成布局图
            self.results = generate_layout_images(
                self.center_images,
                self.side_images,
                self.rib_count,
                self.conf,
                self.symmetry_type
            )

            if not self.results:
                return False, {"error": "未生成任何布局图"}

            # 3. 保存结果
            save_results(self.results, self.conf)

            # 4. 返回成功信息
            return True, {
                "generated_count": len(self.results),
                "symmetry_types": list(set(r.applied_symmetry for r in self.results)),
                "average_score": sum(r.layout_score for r in self.results) / len(self.results)
            }

        except Exception as e:
            return False, {"error": str(e)}
```

### 阶段3: 迁移规则层代码 (land_sea_ratio)

#### 步骤3.1: 创建 rules/scoring/land_sea_ratio.py

**操作**: 从 `code_from_lxl/services/analyzers.py` 提取三个函数

```python
"""
海陆比评分规则模块

评分规则:
- 2分 (优秀): 海陆比在目标范围内 [target_min, target_max]
- 1分 (合格): 海陆比在容差范围内
- 0分 (不合格): 海陆比超出容差范围
"""

import cv2
import numpy as np
from typing import Tuple, Dict, Any


def compute_land_sea_ratio(img: np.ndarray, conf: dict, *args, **kwargs) -> Tuple[int, Dict[str, Any]]:
    """计算输入轮胎花纹样稿的海陆比，并给出评分"""
    # 1. 计算图片总面积
    total_area = img.shape[0] * img.shape[1]

    # 2. 计算黑色和灰色区域面积
    black_area = compute_black_area(img)
    gray_area = compute_gray_area(img)

    # 3. 计算海陆比百分比
    if total_area == 0:
        ratio_percent = 0.0
    else:
        ratio_percent = ((black_area + gray_area) / total_area) * 100

    # 4. 读取评分阈值
    target_min = conf.get("target_min", 28.0)
    target_max = conf.get("target_max", 35.0)
    margin = conf.get("margin", 5.0)

    # 5. 评分逻辑
    if target_min <= ratio_percent <= target_max:
        score = 2
    elif (target_min - margin) <= ratio_percent < target_min or target_max < ratio_percent <= (target_max + margin):
        score = 1
    else:
        score = 0

    # 6. 封装详情
    details = {
        "ratio_value": round(ratio_percent, 2),
        "target_range": f"[{target_min}%, {target_max}%]",
        "black_area": black_area,
        "gray_area": gray_area,
        "total_area": total_area
    }

    return score, details


def compute_black_area(img: np.ndarray, *args, **kwargs) -> int:
    """计算黑色区域面积 (深沟槽, 灰度0-50)"""
    if len(img.shape) == 3:
        gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
    else:
        gray = img

    black_mask = cv2.inRange(gray, 0, 50)
    black_area = cv2.countNonZero(black_mask)
    return black_area


def compute_gray_area(img: np.ndarray, *args, **kwargs) -> int:
    """计算灰色区域面积 (浅沟槽/细花, 灰度51-200)"""
    if len(img.shape) == 3:
        gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
    else:
        gray = img

    gray_mask = cv2.inRange(gray, 51, 200)
    gray_area = cv2.countNonZero(gray_mask)
    return gray_area
```

```python
def _calculate_total_score(task_id: str, conf: dict) -> tuple[bool, dict]:
    """统计总分"""
    from rules.scoring.land_sea_ratio import compute_land_sea_ratio
    from pathlib import Path
    import cv2

    # 1. 确定图片路径
    image_path = conf.get('image_path')
    if not image_path:
        # 默认使用装饰边框后的图片
        base_path = Path(".results") / task_id / "rst"
        if not base_path.exists():
            return False, {"error": f"rst directory not found: {base_path}"}

        # 获取第一个图片文件
        image_files = list(base_path.glob("*.png"))
        if not image_files:
            return False, {"error": "No images found in rst directory"}
        image_path = str(image_files[0])

    # 2. 读取图片
    img = cv2.imread(image_path)
    if img is None:
        return False, {"error": f"Failed to read image: {image_path}"}

    # 3. 计算海陆比评分
    land_sea_conf = conf.get('land_sea_ratio', {})
    land_sea_score, land_sea_details = compute_land_sea_ratio(img, land_sea_conf)

    # 4. 聚合总分
    details = {
        "total_score": land_sea_score,
        "image_path": image_path,
        "scoring_items": {
            "land_sea_ratio": {
                "score": land_sea_score,
                "details": land_sea_details
            }
        }
    }

    return True, details
```
